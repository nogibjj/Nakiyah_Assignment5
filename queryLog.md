CREATE TABLE worker_health (
    Employee_ID NUMERIC PRIMARY KEY,            
    Age INTEGER,                                
    Job_Role TEXT,                              
    Industry TEXT,                              
    Years_of_Experience INTEGER,                
    Work_Location TEXT,                         
    Hours_Worked_Per_Week INTEGER,              
    Mental_Health_Condition TEXT,               
    Access_to_Mental_Health_Resources BOOLEAN);
